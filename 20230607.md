# 파이썬 버전확인
$ python3 -V

# pip 버전확인
$ pip3 -V

# 가상환경 설치 (파이썬버전에 맞춰서 ex 3.8)
$ sudo apt install python3.8-venv

# 가상환경 생성 python3 -m venv [가상환경이름]
$ python3 -m venv venv

# 가상환경 활성
# linux or mac
$ source ./venv/bin/activate

# 가상환경 확인
(venv)$ python -V
(venv)$ pip -V

# 가상환경 비활성
(venv)$ deactivate
